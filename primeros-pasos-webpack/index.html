Primeros pasos con Webpack

Para continuar la introducción a Webpack voy a implementar un proyecto básico.
Primero asegúrate que tienes instalado NodeJS y NPM. Luego en el terminal ejecutamos.

-/app
-main.js
-component.js
-/build
-bundle.js (este se genera automagicamente)
-index.html
-package.json
-webpack.config.js

var path = require(‘path’);

module.exports = {
entry: path.resolve(dirname, ‘app/main’),
output: {
path: path.resolve(dirname, ‘build’),
filename: ‘bundle.js’,
},
};

module.exports = function () {
var element = document.createElement(‘h1’);
element.innerHTML = ‘Hello world’;
return element;
};

var component = require(‘./component’);
var app = document.createElement(‘div’);

document.body.appendChild(app);
app.appendChild(component());
